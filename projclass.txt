-- Create table
CREATE TABLE JSON_DATA_TABLE (
    ID NUMBER,
    JSON_DOCUMENT CLOB CHECK (JSON_DOCUMENT IS JSON)
);

-- Insert valid JSON
INSERT INTO JSON_DATA_TABLE (ID, JSON_DOCUMENT) VALUES (
    1,
    '{
        "EMPLOYEES": [
            {"id": 1, "name": "alice", "department": "cse"},
            {"id": 2, "name": "arun", "department": "hr"}
        ],
        "company_info": {
            "name": "ABC CORP",
            "location": "Bengalore"
        }
    }'
);

-- Query JSON data
SELECT jt.*
FROM JSON_DATA_TABLE d,
     JSON_TABLE(
        d.JSON_DOCUMENT,
        '$.EMPLOYEES[*]'
        COLUMNS (
            EMPLOYEE_ID NUMBER       PATH '$.id',
            EMPLOYEE_NAME VARCHAR2(100) PATH '$.name',
            EMPLOYEE_DEPARTMENT VARCHAR2(100) PATH '$.department'
        )
     ) jt;
